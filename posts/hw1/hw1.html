First Query Function

def query_climate_database(db_file, country, year_begin, year_end, month):
    '''
    Args:
        db_file, the file name for the database
        country, a string giving the name of a country for which data should be returned.
        year_begin and year_end, two integers giving the earliest and latest years for which should be returned (inclusive).
        month, an integer giving the month of the year for which should be returned.
    Returns:
        a Pandas dataframe of temperature readings for the specified country, in the specified date range, in the specified month 
        of the year with these columns: NAME, LATITUDE, LONGITUDE, Country, Year, Month, Temp.
    '''
    #creating the query
    query = '''
    SELECT S.NAME, S.LATITUDE, S.LONGITUDE, C.Name AS Country, T.Year, T.Month, T.Temp
    FROM stations S
    JOIN temperatures T ON S.ID = T.ID
    JOIN countries C ON SUBSTR(T.ID, 1, 2) = C."FIPS 10-4"
    WHERE C.Name = ?
    AND T.Year BETWEEN ? AND ?
    AND T.Month = ?
    ORDER BY S.Name, T.Year
    '''
    #connecting, issuing the query with the correct parameters
    conn = sqlite3.connect(db_file)
    query_df = pd.read_sql_query(query, conn, params=(country, year_begin, year_end, month))
    conn.close()
    return query_df

Geographic Scatter Function for Yearly Temperature Increases

def temperature_coefficient_plot(db_file, country, year_begin, year_end, month, min_obs, **kwargs):
    '''
    Args:
        db_file, the file name for the database
        country, a string giving the name of a country for which data should be returned.
        year_begin and year_end, two integers giving the earliest and latest years for which should be returned (inclusive).
        month, an integer giving the month of the year for which should be returned.
        min_obs, the minimum required number of years of data for any given station
        **kwargs, additional keyword arguments passed to px.scatter_mapbox()
    Returns:
        An interactive geographic scatterplot with a point for each station 
        (the color of the point reflects an estimate of the yearly change in temperature during the specified time period)
    '''
    #creating queried dataframe
    df = query_climate_database(db_file, country, year_begin, year_end, month)

    #ensuring that each year has the minimum number of records in it
    df['year_count'] = df.groupby(['NAME', 'LATITUDE', 'LONGITUDE'])['Year'].transform('nunique')
    df = df[df['year_count'] >= min_obs]

    #coefficient function to find the Linear Regression coefficient for each group
    def coef(data_group):
        X = data_group[["Year"]]
        y = data_group["Temp"]
        LR = LinearRegression()
        LR.fit(X,y)
        slope = LR.coef_[0]
        return slope

    #grouping the data by the location and station
    coefs = df.groupby(['NAME', 'LATITUDE', 'LONGITUDE']).apply(coef).reset_index(name='Estimated Yearly Increase (Â°C)')
    coefs.dropna(subset=['Estimated Yearly Increase (Â°C)'], inplace=True)
    coefs['Estimated Yearly Increase (Â°C)'] = coefs['Estimated Yearly Increase (Â°C)'].round(4)

    #creating the scatter mapbox allowing for any key word arguments
    fig = px.scatter_mapbox(
        coefs,
        lat="LATITUDE",
        lon="LONGITUDE",
        color="Estimated Yearly Increase (Â°C)",
        hover_name="NAME",
        title=f"Estimates of yearly increases in temperature in January for stations in {country}, years {year_begin} - {year_end}", 
        **kwargs, )
    
    return fig

Second Query Function

def query_climate_database_2(db_file, year_begin, year_end):
    '''
    Args:
        db_file, the file name for the database
        year_begin and year_end, two integers giving the earliest and latest years for which should be returned (inclusive).
    Returns:
        a Pandas dataframe of average temperature readings for each country during each month of in the specified year range
        with these columns: Country, Temp, Year, Month
    '''
    #creating the query
    query = '''
    SELECT C.Name AS Country, AVG(T.Temp) as Temp, T.Year, T.Month
    FROM temperatures T
    JOIN countries C ON SUBSTR(T.ID, 1, 2) = C."FIPS 10-4"
    AND T.Year BETWEEN ? AND ?
    GROUP BY C.Name, T.Year, T.Month
    '''
    #connecting, issuing the query with the correct parameters
    conn = sqlite3.connect(db_file)
    query_df = pd.read_sql_query(query, conn, params=(year_begin, year_end))
    conn.close()
    return query_df

Additional Graphs

def temperature_choropleth(db_file, year, **kwargs):
    '''
    Answers the following question: How does the average temperature per country differ in a particular year?
    Args:
        db_file, the file name for the database
        year, integer giving the year for which should be returned
        **kwargs, additional keyword arguments passed to px.scatter()
    Returns:
        An interactive choropleth with a point for each country
        (color is represented by the average temperature at each country)
    '''
    #creating the dataframe with the data
    data = query_climate_database_2(db_file, year, year)
    
    #finding the mean temperature of each country using all of the data
    df = data.groupby("Country")["Temp"].mean().reset_index()    
    df["Temp"] = df["Temp"].round(4)
    
    #plotting the data on a choropleth to see the colors indicating the average temperatures per country
    fig = px.choropleth(
        df,
        locations="Country",
        locationmode = "country names",
        color="Temp",
        title=f"Average Temperature per Country in {year}",
        labels={"Temp": "Average Temp (Â°C)"},
        **kwargs,
    )
    return fig
if __name__ == "__main__":    
    fig = temperature_choropleth("data.db", 2020)
    fig.show()
def temperature_bar_graph(db_file, year_begin, year_end, **kwargs):
    '''
    Answers the following question: How does the average temperature globally per month differ in a range of years?
    Args:
        db_file, the file name for the database
        year, integer giving the year for which should be returned
        **kwargs, additional keyword arguments passed to px.scatter()
    Returns:
        An interactive bar graph with the average global temperatures of each month facetted by year
    '''
    #creating a data frame with the necessary data
    data = query_climate_database_2(db_file, year_begin, year_end)

    #grouping the data by the years and months, then finding the mean temperature for each year-month combination
    df = data.groupby(["Year", "Month"])["Temp"].mean().reset_index()
    df['Year'] = df['Year'].astype(int)
    df["Temp"] = df["Temp"].round(4)

    #mapping names for the numerical month values
    months = {1: "January", 2: "February", 3: "March", 4: "April", 5: "May", 6: "June", 7: "July", 8: "August", 9: "September", 10: "October",  11: "November", 12: "December"}
    df['Months'] = df['Month'].map(months)

    #creating a bar chart
    fig = px.bar(
        df,
        x="Months",
        y="Temp",
        title=f"Average Temperature Across All Countries from {year_begin} to {year_end}",
        labels={"Temp": "Average Temp (Â°C)"},
        facet_col="Year",  # Facet by year
        **kwargs,
    )
    
    return fig
if __name__ == "__main__":
